CURSOR
================================================
SET SERVEROUTPUT ON
DECLARE
	ID INTEGER;	
	Name varchar(20);
	CURSOR c is SELECT C_ID, C_Name from company;
BEGIN
	OPEN c;
	LOOP
		FETCH c into ID, Name;
		dbms_output.put_line(id||' '||name);
		EXIT WHEN c%notfound;
	END LOOP;
	CLOSE c;
END;
/

FUNCTION
================================================
CREATE OR REPLACE FUNCTION Count_Emp 
RETURN number IS 
   total INTEGER; 
BEGIN 
   SELECT count(*) into total FROM company; 
   RETURN total; 
END; 
/ 

TRIGGER
================================================

SET SERVEROUTPUT ON
DECLARE
    c INTEGER;
BEGIN
    c:=Count_Emp();
    dbms_output.put_line('Count: '||c);
END;
/
